RFC3339() { date "+%FT%T%z" | sed "s/\(..\)$/:\1/"; }
_recEscape() { printf %s "${1:-}" | sed "s/\"/\\\\\"/g; s/\t/\\\\t/g; s/$/\\\\n/g" | tr -d "[:cntrl:]" | sed "s/\\\\n$/\n/"; }
_recJSON() { _svr="${1:?}" _msg="$(_recEscape "${2:-}")" && shift 2 && unset _fld _val && for a in "$@"; do if [ "${_val:-}" ]; then _fld="${_fld:?}:\"$(_recEscape "${a:-}")\"" && unset _val && continue; fi && _fld="${_fld:-}${_fld:+,}\"${a:?}\"" _val=1; done && [ $(($# % 2)) = 0 ] || _fld="${_fld:?}:\"\"" && printf "{%s}\n" "\"${REC_TIMESTAMP_KEY:=timestamp}\":\"$(RFC3339)\",\"${REC_SEVERITY_KEY:=severity}\":\"${_svr:?}\",\"${REC_CALLER_KEY:=caller}\":\"$0\",\"${REC_MESSAGE_KEY:=message}\":\"${_msg:?}\"${_fld:+,}${_fld:-}"; }
RecDEFAULT() { test "${REC_SEVERITY:-0}" -gt 000 2>/dev/null || _recJSON DEFAULT "$@"; }
RecDEBUG() { test "${REC_SEVERITY:-0}" -gt 100 2>/dev/null || _recJSON DEBUG "$@"; }
RecINFO() { test "${REC_SEVERITY:-0}" -gt 200 2>/dev/null || _recJSON INFO "$@"; }
RecNOTICE() { test "${REC_SEVERITY:-0}" -gt 300 2>/dev/null || _recJSON NOTICE "$@"; }
RecWARNING() { test "${REC_SEVERITY:-0}" -gt 400 2>/dev/null || _recJSON WARNING "$@"; }
RecERROR() { test "${REC_SEVERITY:-0}" -gt 500 2>/dev/null || _recJSON ERROR "$@"; }
RecCRITICAL() { test "${REC_SEVERITY:-0}" -gt 600 2>/dev/null || _recJSON CRITICAL "$@"; }
RecALERT() { test "${REC_SEVERITY:-0}" -gt 700 2>/dev/null || _recJSON ALERT "$@"; }
RecEMERGENCY() { test "${REC_SEVERITY:-0}" -gt 800 2>/dev/null || _recJSON EMERGENCY "$@"; }
_recArg() { for a in "$@"; do if echo "${a:-}" | grep -Eq "[[:blank:]]"; then printf "'%s' " "${a:-}"; else printf "%s " "${a:-}"; fi; done | sed "s/ $//"; }
RecExec() { RecDEBUG "$ $(_recArg "$@")" && "$@"; }
RecRun() { _dlm='####R#E#C#D#E#L#I#M#I#T#E#R####' _all=$({ _out=$("$@") && _rtn=$? || _rtn=$? && printf "\n%s" "${_dlm:?}${_out:-}" && return ${_rtn:-0}; } 2>&1) && _rtn=$? || _rtn=$? && _dlmno=$(echo "${_all:-}" | sed -n "/${_dlm:?}/=") && _cmd=$(_recArg "$@") && _stdout=$(echo "${_all:-}" | tail -n +"${_dlmno:-1}" | sed "s/^${_dlm:?}//") && _stderr=$(echo "${_all:-}" | head -n "${_dlmno:-1}" | grep -v "^${_dlm:?}") && RecDEBUG "$ ${_cmd:-}" command "${_cmd:-}" stdout "${_stdout:-}" stderr "${_stderr:-}" return "${_rtn:-0}" && return ${_rtn:-0}; }
